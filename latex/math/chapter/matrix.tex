\section{The Cross Product}

\begin{equation}
  \va{u} \cross \va{v} =
  \underbrace{
    \begin{vmatrix}
      \vu{i} & \vu{j} & \vu{k} \\
      u_1    & u_2    & u_3    \\
      v_1    & v_2    & v_3
    \end{vmatrix}
  }_{\text{Determinant}} =
  \begin{vmatrix}
    u_2 & u_3 \\
    v_2 & v_3
  \end{vmatrix} \uvec{i} -
  \begin{vmatrix}
    u_3 & u_1 \\
    v_3 & v_1
  \end{vmatrix} \uvec{j} +
  \begin{vmatrix}
    u_1 & u_2 \\
    v_1 & v_2
  \end{vmatrix} \uvec{k}
\end{equation}
